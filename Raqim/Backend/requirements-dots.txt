# dots.mocr (DotOCR) dependencies — branch-scoped.
#
# Used by both the standalone PoC (test_dots_ocr.py) and the in-app DotOCR
# layout engine (dots_ocr_engine.py, enabled via OCR_ENGINE=dots).
#
# These are kept SEPARATE from the main requirements.txt because the two OCR
# engines need DIFFERENT, mutually-exclusive transformers versions:
#   * DeepSeek-OCR-2 (default) -> transformers==4.46.3 (requirements.txt)
#   * DotOCR / dots.mocr       -> transformers==4.57.6 (this file)
# Only one engine can be installed in a given environment at a time.
#
# For the in-app DotOCR engine you also need the base app deps (Flask,
# pdf2image, python-docx, Pillow ...) from requirements.txt, but with
# transformers pinned to 4.57.6 (install requirements.txt first, then this file).
#
# Standalone PoC install into an isolated environment, e.g.:
#
#   cd Raqim/Backend
#   python3 -m venv venv-dots
#   source venv-dots/bin/activate
#   # Install a torch build that matches your platform first. For CUDA, e.g.:
#   #   pip install torch==2.7.0 torchvision==0.22.0 --index-url https://download.pytorch.org/whl/cu128
#   pip install -r requirements-dots.txt
#   python test_dots_ocr.py path/to/page.png
#
# A CUDA GPU is strongly recommended; CPU inference works but is very slow.

# Core deep-learning / vision-language stack.
torch>=2.5.0
torchvision>=0.20.0

# dots.mocr ships its model code via trust_remote_code on top of transformers.
# The upstream repo pins transformers==4.57.6. This pin is REQUIRED for correct
# results: on transformers 5.x the model still loads and runs, but its remote
# code is incompatible with the refactored base/generation classes and emits
# empty/incorrect OCR output. Do not bump this without re-verifying output.
transformers==4.57.6
accelerate>=0.34.0

# Image/video preprocessing helper used by the dots.mocr / Qwen2-VL processor.
qwen-vl-utils>=0.0.8

# Image IO.
Pillow>=10.1.0

# Optional, GPU-only, large build. Enables faster attention if available.
# The test script automatically falls back to sdpa/eager when this is absent.
# flash-attn>=2.8.0
